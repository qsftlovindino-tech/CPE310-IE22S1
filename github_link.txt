def get_director_titles(df, director_name):

    mask = df["director"].str.contains(director_name, case=False, na=False)
    result = df[mask].copy()
    return result.sort_values(by="release_year", ascending=True)

director_df = get_director_titles(df, "Spielberg")
display(director_df[["title", "director", "release_year"]].head(10))

"""b) Generic plotting function: top N categories"""

def plot_top_categories(df, column, n=10, figsize=(10,5)):

    counts = df[column].value_counts().head(n)

    plt.figure(figsize=figsize)
    sns.barplot(x=counts.values, y=counts.index, orient="h")
    plt.title(f"Top {n} categories in '{column}'")
    plt.xlabel("Count")
    plt.ylabel(column)
    plt.tight_layout()
    plt.show()

plot_top_categories(df, "listed_in", n=10)
plot_top_categories(df, "country", n=10)
plot_top_categories(df, "rating", n=10)
